06. 오답노트 - 5보다 큰 수 구하기

const num = [5, 9, 7, 10];

function solution(arr){
    let sm = arr[0], index = 0;
    for(i=0; i < arr.length; i++){
        if(arr[i] < sm){
            sm = arr[i];
            index = i;
        }
    }
    arr.splice(index, 1);
    return arr;
}
console.log(solution(num));     //[9, 7, 10]
