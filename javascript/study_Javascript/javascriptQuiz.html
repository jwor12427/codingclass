09. 결과값을 작성하세요.

const num = [4,3,1,6];

function solution(s){
    let answer = [];
    for(let i = 0; i < s.length; i++){
        for(let j = i+1; j < s.length; j++){
            const result = s[i] + s[j];
            if(answer.indexOf(result) < 1){
                answer.push(result);
            }
        }
    }
    answer.sort((a,b) => a - b);
    return answer;
}
console.log(solution(num));     //[4, 5, 7, 7, 9, 10]
