document.querySelectorAll(".content__item__desc.reveal").forEach(text => {
    spanText = text.innerText;
    span = "" + spanText + "";
    text.innerHTML = span;
});

function scroll(){
    let scrollTop = window.scrollY;

    const reveal = document.querySelectorAll(".reveal");
    reveal.forEach(el => {
        let revealOffset = el.offsetTop + el.parentElement.offsetTop;   //부모 박스의 값을 가져와야하고, reveal이 붙은 이미지박스와 부모 박스 간의 간격값도 가져와야함
        let revealDelay = el.dataset.delay;


        // if(scrollTop > revealOffset - window.innerHeight/2){           //reveal이 붙어 있는 곳의 offsetTop값을 구해야함
        //     // console.log(revealOffset);    
        //     el.classList.add("show");
        // }       
        
        if(scrollTop >= revealOffset - window.innerHeight){
            if(revealDelay == undefined){
                el.classList.add("show");
            } else {
                setTimeout(() => {
                    el.classList.add("show");
                },revealDelay);
            }
        }
    })
    requestAnimationFrame(scroll);
}
scroll();
